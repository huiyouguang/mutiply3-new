const darkMode = dv.metaBind?.get('obsidian-theme-dark') ?? false;
const theme = darkMode ? {
    bg: '#333338', text: '#e6e6e6', accent: '#b86a7c'
} : {
    bg: '#f5edeb', text: '#444', accent: '#d4889a'
};

const today = new Date();
today.setHours(0,0,0,0);

// 今日新建笔记
const todayPages = dv.pages().filter(p => p.file.cday >= today);
const noteCount = todayPages.length;

// 异步读取所有今日笔记的内容并统计字数
let wordCount = 0;
const promises = todayPages.map(async (p) => {
    // 使用 Dataview 的 io 模块异步加载文件内容
    const content = await dv.io.load(p.file.path);
    if (!content) return 0;
    // 只统计汉字、字母数字等有效字符（可根据需求调整）
    const pure = content.replace(/[\s\n\r#\*\-_>\[\]\(\)]/g, '');
    return pure.length;
});

const wordCounts = await Promise.all(promises);
wordCount = wordCounts.reduce((a, b) => a + b, 0);

// 今日完成任务
let doneTask = 0;
for(let p of dv.pages()){
    if(!p.file.tasks) continue;
    for(let t of p.file.tasks){
        if(t.completed && t.completion >= today) doneTask++;
    }
}

// 渲染卡片（放在最后，确保异步数据已拿到）
dv.paragraph(`
<div style="padding:14px 18px;background:${theme.bg};border-radius:14px;margin-bottom:20px;color:${theme.text};">
📊 今日统计
✅ 新建笔记：<strong style="color:${theme.accent};">${noteCount}</strong> 篇
｜✍️ 今日字数：<strong style="color:${theme.accent};">${wordCount}</strong> 字
｜🎯 完成任务：<strong style="color:${theme.accent};">${doneTask}</strong> 个
</div>
`);
