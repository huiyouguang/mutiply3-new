```dataviewjs
// 今日新建笔记
const todayPages = dv.pages().filter(p => p.file.cday >= today);
const noteCount = todayPages.length;

// 异步读取所有今日笔记的内容并统计字数
let wordCount = 0;
const promises = todayPages.map(async (p) => {
    // 使用 Dataview 的 io 模块异步加载文件内容
    const content = await dv.io.load(p.file.path);
    if (!content) return 0;
    // 只统计汉字、字母数字等有效字符（可根据需求调整）
    const pure = content.replace(/[\s\n\r#\*\-_>\[\]\(\)]/g, '');
    return pure.length;
});

const wordCounts = await Promise.all(promises);
wordCount = wordCounts.reduce((a, b) => a + b, 0);

// 今日完成任务
let doneTask = 0;
for(let p of dv.pages()){
    if(!p.file.tasks) continue;
    for(let t of p.file.tasks){
        if(t.completed && t.completion >= today) doneTask++;
    }
}

// 渲染卡片（放在最后，确保异步数据已拿到）
dv.paragraph(`
<div style="padding:14px 18px;background:${theme.bg};border-radius:14px;margin-bottom:20px;color:${theme.text};">
```

```dataviewjs

// ========== Thino 随机灵感 - 全版本兼容 终极稳定版 ==========
const THINO_FOLDER = "thino";
const JOURNAL_HEADING = "✏随笔感悟";
const RANDOM_COUNT = 5;

// 深浅主题适配
const darkMode = dv.metaBind?.get('obsidian-theme-dark') ?? false;
const theme = darkMode ? {
    bg: '#2a2a2e',
    text: '#e6e6e6',
    borderColor: '#b86a7c',
    buttonBg: '#b86a7c'
} : {
    bg: '#fefaf5',
    text: '#444',
    borderColor: '#d4889a',
    buttonBg: '#d4889a'
};

// 获取内容
const pages = dv.pages(`"${THINO_FOLDER}"`).filter(p => p.file.ext === "md");
let allInspirations = [];

for (let page of pages) {
    const lists = page.file.lists;
    if (!lists) continue;
    for (let list of lists) {
        const section = list.section?.subpath || "";
        if (!section.includes(JOURNAL_HEADING) && !section.includes("随笔感悟")) continue;
        
        let rawText = list.text;
        if (!rawText) continue;
        if (/^\s*\d{1,2}:\d{2}\s*$/.test(rawText.trim())) continue;
        
        let cleanText = rawText.replace(/^\s*\d{1,2}:\d{2}\s*/, '').trim();
        cleanText = cleanText.replace(/@completed\([^)]*\)\s*/g, '').trim();
        if (/^✅\s*\d{4}-\d{2}-\d{2}(\/\d{2}:\d{2})?\s*$/.test(cleanText)) continue;
        cleanText = cleanText.replace(/^✅\s*\d{4}-\d{2}-\d{2}(\/\d{2}:\d{2})?\s*/, '').trim();
        if (cleanText === "") continue;

        // 安全转义
        cleanText = cleanText.replace(/</g, '&lt;').replace(/>/g, '&gt;');
        const link = `<a href="${page.file.path}" style="color:${theme.borderColor};text-decoration:none;">🔗 原文</a>`;

        allInspirations.push({
            text: cleanText,
            link: link,
            date: page.file.name
        });
    }
}

// 输出标题 + 按钮
dv.paragraph(`
<div style="display: flex; justify-content: space-between; align-items: center; margin-bottom: 18px;">
    <h3 style="margin:0; color: ${theme.text};">✨ Thino 随机灵感</h3>
    <button class="random-refresh-btn" style="background: ${theme.buttonBg}; border: none; color: white; padding: 5px 16px; border-radius: 30px; font-size: 13px; cursor: pointer;">🍃 换一批</button>
</div>
`);

// 输出卡片容器
dv.paragraph(`<div id="thino-cards" style="width: 100%;"></div>`);

// 等待DOM加载后渲染
setTimeout(() => {
    const container = document.getElementById('thino-cards');
    if (!container) return;

    // 渲染函数
    function render() {
        const shuffled = [...allInspirations].sort(() => Math.random() - 0.5);
        const selected = shuffled.slice(0, RANDOM_COUNT);
        let html = '';
        for (let q of selected) {
            html += `
<div style="margin-bottom: 14px; padding: 12px 16px; background: ${theme.bg}; border-left: 4px solid ${theme.borderColor}; border-radius: 10px;">
    <div style="color: ${theme.text}; line-height:1.6;">“${q.text}”</div>
    <div style="font-size: 12px; color: #888; margin-top: 8px; text-align: right;">
        📅 ${q.date} · ${q.link}
    </div>
</div>
            `;
        }
        container.innerHTML = html;
    }

    // 首次渲染
    render();

    // 按钮绑定事件
    document.querySelector('.random-refresh-btn')?.addEventListener('click', render);
}, 100);
```

## 3 🔍 智能检索

### 3.1 🏷️ 热门标签
```dataviewjs
// 获取暗色模式配置
const darkMode = dv.metaBind?.get('obsidian-theme-dark') ?? false;
const theme = darkMode ? {
    bg: '#333338', text: '#e6e6e6', accent: '#b86a7c', tagBg: '#4a4a55'
} : {
    bg: '#f5edeb', text: '#444', accent: '#d4889a', tagBg: '#e8dedb'
};

// 统计标签（排除 project/ 和 主页/ 开头的标签）
const tagCount = {};
for (let page of dv.pages()) {
    const tags = page.file.tags;
    if (tags) {
        for (let tag of tags) {
            if (!tag.startsWith("project/") && !tag.startsWith("主页/")) {
                tagCount[tag] = (tagCount[tag] || 0) + 1;
            }
        }
    }
}

// 排序并取前20
const sortedTags = Object.entries(tagCount)
    .sort((a, b) => b[1] - a[1])
    .slice(0, 20);

// 生成 HTML（可点击的标签）
let html = `
<div style="padding:14px 18px; background:${theme.bg}; border-radius:14px; margin-bottom:20px;">
<div style="margin-bottom:12px; color:${theme.text}; font-weight:600;">🏷️ Top 20 热门标签</div>
<div style="display:flex; flex-wrap:wrap; gap:10px;">`;

for (let [tag, count] of sortedTags) {
    // 转义标签中的引号
    const escapedTag = tag.replace(/'/g, "\\'");
    // 标签可点击：搜索该标签
    html += `<a href="#" onclick="app.workspace.openLinkText('tag:${escapedTag}', '', false); return false;" 
                style="display:inline-block; background:${theme.tagBg}; color:${theme.text}; 
                       padding:6px 12px; border-radius:20px; text-decoration:none; font-size:0.9em;">
                #${tag} <span style="color:${theme.accent};">(${count})</span>
             </a>`;
}

html += `</div></div>`;
dv.paragraph(html);
```
