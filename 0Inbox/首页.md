## 0.3 📊 全局统计

```dataviewjs
const totalNotes = dv.pages().length;
const today = dv.date('today');
const todayEdits = dv.pages().filter(p => p.file?.mtime && p.file.mtime >= today).length;

const allTags = new Set();
dv.pages().forEach(p => {
    if (p.file?.tags) {
        p.file.tags.forEach(t => allTags.add(t));
    }
});

const taskFolderPages = dv.pages('"01 task"');
const allTasksInFolder = taskFolderPages.file.tasks;

const targetSubjects = [
    { tag: "#注册税法", name: "注册税法" },
    { tag: "#注册经济法", name: "注册经济法" },
    { tag: "#注册财管", name: "注册财务成本管理" },
    { tag: "#注册审计", name: "注册审计" },
    { tag: "#注册战略", name: "注册公司战略与风险管理" },
    { tag: "#中级实务", name: "中级会计" },
    { tag: "#中级经济法", name: "中级经济法" },
    { tag: "#中级财管", name: "中级财务管理" }
];

const subjectStats = {};
targetSubjects.forEach(subj => {
    subjectStats[subj.tag] = { name: subj.name, total: 0, completed: 0, uncompleted: 0, rate: "0%" };
});

let globalTotal = 0, globalCompleted = 0, globalUncompleted = 0;

allTasksInFolder.forEach(task => {
    if (!task || !task.text) return;
    const isCompleted = task.completed;
    const taskText = task.text;
    globalTotal++;
    if (isCompleted) globalCompleted++; else globalUncompleted++;
    targetSubjects.forEach(subj => {
        if (taskText.includes(subj.tag)) {
            subjectStats[subj.tag].total++;
            if (isCompleted) subjectStats[subj.tag].completed++;
            else subjectStats[subj.tag].uncompleted++;
        }
    });
});

targetSubjects.forEach(subj => {
    const stats = subjectStats[subj.tag];
    stats.rate = stats.total > 0 ? ((stats.completed / stats.total) * 100).toFixed(1) + '%' : '0%';
});
const globalRate = globalTotal > 0 ? ((globalCompleted / globalTotal) * 100).toFixed(1) + '%' : '0%';

dv.paragraph(`📝 **总笔记数：** ${totalNotes} | ✨ **今日编辑：** ${todayEdits} | 🏷️ **标签数：** ${allTags.size} | 📅 **今日：** ${moment().format('YYYY-MM-DD dddd')}`);

dv.header(4, '📈 全局任务进度');
dv.paragraph(`**任务总数：** ${globalTotal} 📋 | **已完成：** ${globalCompleted} ✅ | **未完成：** ${globalUncompleted} ⏳ | **完成率：** ${globalRate} 📈`);
```
